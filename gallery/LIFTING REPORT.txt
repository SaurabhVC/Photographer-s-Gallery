FORM HTML_TOP_OF_PAGE USING R_TOP TYPE REF TO CL_DD_DOCUMENT.

  DATA: S_TAB   TYPE SDYDO_TEXT_TABLE,
        C1_AREA TYPE REF TO CL_DD_AREA,
        C2_AREA TYPE REF TO CL_DD_AREA,
        TEXT    TYPE SDYDO_TEXT_ELEMENT.



  DATA :
    DAY(2)         TYPE C,
    MONTH(2)       TYPE C,
    YEAR(2)        TYPE C,
    MONTH_WORDS(3) TYPE C,
    DATE1(9)       TYPE C,
    DATE2(9)       TYPE C,
    DATE3(9)       TYPE C,
    TIME(10)       TYPE C,
    HRS(2)         TYPE C,
    MIN(2)         TYPE C,
    SEC(2)         TYPE C,
    ADRNR          TYPE TVKO-ADRNR,
    NAME           TYPE CHAR35.


  DAY = DATE-LOW+6(2).
  MONTH = DATE-LOW+4(2).
  YEAR = DATE-LOW+2(2).

  HRS = SY-UZEIT+0(2).
  MIN = SY-UZEIT+2(2).
  SEC = SY-UZEIT+4(2).
  CONCATENATE HRS MIN SEC INTO TIME SEPARATED BY ':'.
* Month name from the table T247
  SELECT SINGLE KTX
  INTO MONTH_WORDS
  FROM T247
  WHERE SPRAS = SY-LANGU
  AND MNR = MONTH.

  CONCATENATE DAY MONTH_WORDS YEAR INTO DATE1 SEPARATED BY '-'.

  CLEAR : DAY, MONTH, YEAR, MONTH_WORDS.

  DAY = DATE-HIGH+6(2).
  MONTH = DATE-HIGH+4(2).
  YEAR = DATE-HIGH+2(2).
* Month name from the table T247
  SELECT SINGLE KTX
  INTO MONTH_WORDS
  FROM T247
  WHERE SPRAS = SY-LANGU
  AND MNR = MONTH.

  CONCATENATE DAY MONTH_WORDS YEAR INTO DATE2 SEPARATED BY '-'.

  CLEAR : DAY, MONTH, YEAR, MONTH_WORDS.

  DAY = SY-DATUM+6(2).
  MONTH = SY-DATUM+4(2).
  YEAR = SY-DATUM+2(2).
* Month name from the table T247
  SELECT SINGLE KTX
  INTO MONTH_WORDS
  FROM T247
  WHERE SPRAS = SY-LANGU
  AND MNR = MONTH.

  CONCATENATE MONTH_WORDS DAY YEAR INTO DATE3 SEPARATED BY '-'.

  SELECT SINGLE ADRNR FROM TVKO INTO ADRNR WHERE VKORG = SALESORG.
  SELECT SINGLE NAME1 FROM ADRC INTO NAME WHERE ADDRNUMBER = ADRNR.


  CALL METHOD R_TOP->INITIALIZE_DOCUMENT.
*------------------------------------------------
  CALL METHOD R_TOP->HORIZONTAL_SPLIT
    EXPORTING
      SPLIT_AREA = R_TOP
    IMPORTING
      BELOW_AREA = C2_AREA.
*-------------------------------------------------------
*CALL METHOD R_TOP->NEW_LINE.
*CALL METHOD R_TOP->NEW_LINE.

  READ TABLE ITAB INTO WA INDEX 1.

  CONCATENATE 'SHIPPING IMPORT' ':' 'LIFTING REPORT' INTO TEXT SEPARATED BY SPACE.


  CALL METHOD C2_AREA->ADD_TEXT
    EXPORTING
      TEXT         = TEXT
      SAP_EMPHASIS = 'HEAD'.
  CLEAR TEXT.

  CALL METHOD C2_AREA->NEW_LINE.
  CALL METHOD C2_AREA->NEW_LINE.

  CONCATENATE DATE3 TIME INTO TEXT SEPARATED BY SPACE.
  CALL METHOD C2_AREA->ADD_TEXT
    EXPORTING

      SAP_EMPHASIS = 'STRONG'.
  CLEAR TEXT.

  CALL METHOD C2_AREA->NEW_LINE.
  CALL METHOD C2_AREA->NEW_LINE.

  CALL METHOD C2_AREA->ADD_GAP
    EXPORTING
      WIDTH = 80.

  CALL METHOD C2_AREA->ADD_TEXT
    EXPORTING
      TEXT         = TEXT
      SAP_EMPHASIS = 'STRONG'.
  CLEAR TEXT.
